(function() {

  'use strict';

  var resizeTable = {};

  resizeTable = {

    def: {
      offset: 0,
      thActive: undefined,
      thElmnts: document.getElementsByTagName('th'),
      resizerEl: document.createElement('span'),
      resizerElPrev: document.createElement('span')
    },

    /**
     * @function
     * @description: attach resize handles to TH cell
     * @input: HTML Element [<th>]
     */
    mouseenterCall: function(ele) {
      ele.addEventListener('mouseenter', function(e) {
        e.target.appendChild(resizeTable.def.resizerEl);
        if (e.target.previousElementSibling) {
          e.target.previousElementSibling.appendChild(resizeTable.def.resizerElPrev);
        }
      });
    },

    /**
     * @function
     * @description: remove previously attached handle elements to TH cell
     * @input: HTML Element [<th>]
     */
    mouseleaveCall: function(ele) {
      ele.addEventListener('mouseleave', function(e) {
        resizeTable.def.resizerEl.remove();
        resizeTable.def.resizerElPrev.remove();
      });
    },

    /**
     * @function
     * @description: define beginning offset X and TH node
     * @input: HTML Element [<th>]
     */
    mousedownCall: function(ele) {
      ele.addEventListener('mousedown', function(e) {
        resizeTable.def.offset = e.target.parentNode.offsetWidth - e.pageX;
        resizeTable.def.thActive = e.target.parentNode;
      });
    },

    /**
     * @function
     * @description: remove defined TH node, this will stop excessive mousemove calculations
     */
    mouseupCall: function() {
      document.addEventListener('mouseup', function(e) {
        resizeTable.def.thActive = null;
      });
    },

    /**
     * @function
     * @description: calculate offset and apply the right width to the active TH element
     */
    mousemoveCall: function() {
      document.addEventListener('mousemove', function(e) {
        if (typeof resizeTable.def.thActive === 'undefined' || !resizeTable.def.thActive) {
          return;
        }
        resizeTable.def.thActive.style.width = resizeTable.def.offset + e.pageX + 'px';
      });
    },

    /**
     * @function
     * @description: main initialization
     */
    init: function() {
      for (var i = 0, len = resizeTable.def.thElmnts.length; i < len; i++) {
        var iThEle = resizeTable.def.thElmnts[i];

        resizeTable.mouseenterCall(iThEle);
        resizeTable.mouseleaveCall(iThEle);
        resizeTable.mousedownCall(iThEle);
      }

      resizeTable.mouseupCall();
      resizeTable.mousemoveCall();
    }
  };

  resizeTable.init();

})();
